Node.js - Chat API 
const axios = require('axios');

async function chat(message) {
  const response = await axios.post('https://lumenchat.org/api/chat', {
    message: message,
    sessionId: 'SESSION_ID_PLACEHOLDER',
    autoApprove: true
  });
  
  console.log('Response:', response.data.response.response);
}

chat('What is machine learning?');
Python - Chat API 
import requests

def chat(message):
    response = requests.post('https://lumenchat.org/api/chat', json={
        'message': message,
        'sessionId': 'SESSION_ID_PLACEHOLDER',
        'autoApprove': True
    })
    
    data = response.json()
    print(f"Response: {data['response']['response']}")

chat('What is machine learning?')

Browser JavaScript - Fetch API 
async function chat(message) {
  const response = await fetch('https://lumenchat.org/api/chat', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      message: message,
      sessionId: 'SESSION_ID_PLACEHOLDER'
    })
  });
  
  const data = await response.json();
  console.log('Response:', data);
}

chat('What is machine learning?');
WebSocket - Terminal Streaming 
const ws = new WebSocket('wss://lumenchat.org/ws');

ws.onopen = () => {
  ws.send(JSON.stringify({
    type: 'execute',
    command: 'npm install express',
    sessionId: 'SESSION_ID_PLACEHOLDER'
  }));
};

ws.onmessage = (event) => {
  const msg = JSON.parse(event.data);
  
  if (msg.type === 'stdout') {
    console.log(msg.data);
  } else if (msg.type === 'complete') {
    console.log('Done! Exit code:', msg.result.exitCode);
  }
};
